Floating Point Subtraction in C

Code:

#include <stdio.h>

void converter(float num, float *number, int *exponent) {
    *number = num;
    *exponent = 0;

    while ((int)(*number) != 0) {
        *number /= 10;
        (*exponent)++;
    }
}

void subtractor(float first, float second, float *result_number, int *result_exponent) {
    float p1_number, p2_number;
    int p1_exponent, p2_exponent;

    converter(first, &p1_number, &p1_exponent);
    converter(second, &p2_number, &p2_exponent);

    while (p1_exponent != p2_exponent) {
        if (p1_exponent < p2_exponent) {
          p1_number /= 10;
          p1_exponent++;
        }
        else {
          p2_number /= 10;
          p2_exponent++;
        }
    }

    *result_number = p1_number - p2_number;
    *result_exponent = p1_exponent;
}

int main() {
    float first, second, result_number;
    int result_exponent;

    printf("Enter two floating point numbers (a b): ");
    scanf("%f %f", &first, &second);

    subtractor(first, second, &result_number, &result_exponent);

    printf("%.6f - %.6f = %.6fE%d\n", first, second, result_number, result_exponent);

    return 0;
}
