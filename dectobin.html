Decimal to Binary Conversion in C++

Code:

// Decimal to Binary Conversion
#include <iostream>
using namespace std;
int main() {
  int num, i, j, rem[50], len = 0;
  cout << "Enter a decimal number: ";
  cin >> num;
  for (i = 0; num > 1; i++) {
    rem[i] = num % 2;
    num = num / 2;
    len++;
  }
  rem[i] = num;
  for (j = i; j >= 0; j--) {
    cout << rem[j];
  }
  return 0;
}

Output:

Enter a decimal number: 17
10001
