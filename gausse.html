Solving Linear Equations Using Gauss Elimination Method in C

Code:

#include <stdio.h>
#define N 3

void gaussElimination(float matrix[N][N + 1], float result[N]) {
    int i, j, k;
    float factor;
    for (i = 0; i < N - 1; i++) {
        for (j = i + 1; j < N; j++) {
            factor = matrix[j][i] / matrix[i][i];
            for (k = i; k < N + 1; k++) {
                matrix[j][k] -= factor * matrix[i][k];
            }
        }
    }
    for (i = N - 1; i >= 0; i--) {
        result[i] = matrix[i][N];
        for (j = i + 1; j < N; j++) {
            result[i] -= matrix[i][j] * result[j];
        }
        result[i] /= matrix[i][i];
    }
}

int main() {
    float matrix[N][N + 1], result[N];
    int i, j;
    printf("Enter the Coefficient Matrix\n");
    for (i = 0; i < N; i++) {
        for (j = 0; j < N; j++) {
            printf("%c%d : ", 97+j, i+1);
            scanf("%f", &matrix[i][j]);
        }
    }
    printf("Enter the Constant Matrix\n");
    for (i = 0; i < N; i++) {
        printf("%c%d : ", 97+i, i+1);
        scanf("%f", &matrix[i][N]);
    }
    gaussElimination(matrix, result);
    printf("Solution:\n");
    for (i = 0; i < N; i++) {
        printf("x%d = %f\t", i + 1, result[i]);
    }
    return 0;
}
