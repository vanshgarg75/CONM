Binary to Decimal Conversion in C++

Code:

// Binary to Decimal Conversion
#include <iostream>
#include <math.h>
using namespace std;
int main() {
  int bin, i, dec = 0, x, y, num;
  cout << "Enter binary number: ";
  cin >> bin;
  num = bin;
  for (i = 0; bin >= 1; i++) {
    x = bin % 10;
    bin = bin / 10;
    y = x * pow(2, i);
    dec = dec + y;
  }
  cout << "Decimal value of " << num << " is " << dec;
}

Enter binary number: 1010
